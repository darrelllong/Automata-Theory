%% 7.1
\item \begin{enumerate}[(a)]
  \item \textbf{Prove $\OBAR(t,yx)=\OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x)$} for any Mealy machine.

  \begin{proof}
  By induction on $|x|=k$.

  \emph{Base ($k=0$):} $\OBAR(t,y\lambda)=\OBAR(t,y)=\OBAR(t,y)\cdot\lambda=\OBAR(t,y)\cdot\OBAR(\DBAR(t,y),\lambda)$ since $\OBAR(\cdot,\lambda)=\lambda$.

  \emph{Step:}  Let $x=x'\pmb{a}$ with $|x'|=k$.  Then:
  \begin{align*}
    \OBAR(t,yx'\pmb{a})
    &= \OBAR(t,yx')\cdot\omega(\DBAR(t,yx'),\pmb{a}) \\
    &= \OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x')\cdot\omega(\DBAR(\DBAR(t,y),x'),\pmb{a}) \\
    &\quad\text{(inductive hypothesis applied to $yx'$)} \\
    &= \OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x'\pmb{a}) \\
    &= \OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x). \qed
  \end{align*}
  \end{proof}

  \item \textbf{Prove $\DBAR(s,yx)=\DBAR(\DBAR(s,y),x)$.}

  \begin{proof}
  By induction on $|x|$.

  \emph{Base:} $\DBAR(s,y\lambda)=\DBAR(s,y)=\DBAR(\DBAR(s,y),\lambda)$.

  \emph{Step:}  $\DBAR(s,yx'\pmb{a})=\delta(\DBAR(s,yx'),\pmb{a})=\delta(\DBAR(\DBAR(s,y),x'),\pmb{a})=\DBAR(\DBAR(s,y),x'\pmb{a})$. \qed
  \end{proof}
\end{enumerate}

%% 7.2
\item Let $\mu:S_A\to S_B$ be a homomorphism between Mealy machines.

\begin{enumerate}[(a)]
  \item \textbf{Prove $\mu(\DBAR_A(s,x))=\DBAR_B(\mu(s),x)$.}

  \begin{proof}
  By induction on $|x|$.

  \emph{Base:} $\mu(\DBAR_A(s,\lambda))=\mu(s)=\DBAR_B(\mu(s),\lambda)$.

  \emph{Step:}  $\mu(\DBAR_A(s,x\pmb{a}))=\mu(\delta_A(\DBAR_A(s,x),\pmb{a}))=\delta_B(\mu(\DBAR_A(s,x)),\pmb{a})=\delta_B(\DBAR_B(\mu(s),x),\pmb{a})=\DBAR_B(\mu(s),x\pmb{a})$.
  (Used: $\mu(\delta_A(t,\pmb{a}))=\delta_B(\mu(t),\pmb{a})$ for all $t\in S_A$---homomorphism property.) \qed
  \end{proof}

  \item \textbf{Prove $\OBAR_A(s,x)=\OBAR_B(\mu(s),x)$.}

  \begin{proof}
  By induction on $|x|$.

  \emph{Base:} $\OBAR_A(s,\lambda)=\lambda=\OBAR_B(\mu(s),\lambda)$.

  \emph{Step:}  $\OBAR_A(s,x\pmb{a})=\OBAR_A(s,x)\cdot\omega_A(\DBAR_A(s,x),\pmb{a})$.
  By induction: $=\OBAR_B(\mu(s),x)\cdot\omega_A(\DBAR_A(s,x),\pmb{a})$.
  By homomorphism: $\omega_A(t,\pmb{a})=\omega_B(\mu(t),\pmb{a})$ for all $t$; so:
  $=\OBAR_B(\mu(s),x)\cdot\omega_B(\mu(\DBAR_A(s,x)),\pmb{a})=\OBAR_B(\mu(s),x)\cdot\omega_B(\DBAR_B(\mu(s),x),\pmb{a})=\OBAR_B(\mu(s),x\pmb{a})$. \qed
  \end{proof}
\end{enumerate}

%% 7.3
\item \textbf{Prove Corollary~7.3.}

Corollary~7.3 states (typically): if there is a homomorphism $\mu:A\to B$ between connected Mealy machines, then $B$ is a homomorphic image of $A$ and $A$ is equivalent to $B$.

\begin{proof}
By Exercise~7.2(b), $\OBAR_A(s_0,x)=\OBAR_B(\mu(s_0),x)$ for all $x\in\Sigma^*$.  Since $\mu(s_{0_A})=s_{0_B}$ (by definition of homomorphism), $\OBAR_A(s_{0_A},x)=\OBAR_B(s_{0_B},x)$, i.e., $f_A(x)=f_B(x)$ for all $x$.  Hence $A\equiv B$. \qed
\end{proof}

%% 7.4
\item \textbf{Prove Corollary~7.4: $M$ is minimal iff $M$ is reduced and connected.}

\begin{proof}
($\Rightarrow$) If $M$ is minimal, it is the smallest machine computing $f_M$.  Any disconnected state can be removed without changing $f_M$ (since the machine starts at $s_0$ and disconnected states are unreachable).  Any reducible machine (with two equivalent states) can be merged.  Both would yield a smaller equivalent machine, contradicting minimality.  Hence $M$ is reduced and connected.

($\Leftarrow$) If $M$ is reduced and connected, assume for contradiction there is an equivalent machine $M'$ with fewer states.  By Theorem~7.5, the minimal machine is unique (up to isomorphism); since $M$ is reduced and connected, it has no redundant states.  If $M'$ has fewer states, some states of $M$ must correspond to multiple states of $M'$ via any homomorphism, but this contradicts $M$ being reduced.  Hence $M$ is minimal. \qed
\end{proof}

%% 7.5
\item \textbf{Pumping lemma for FTD functions.}

\begin{enumerate}[(a)]
  \item \textbf{Statement:}  Let $f:\Sigma^*\to\Gamma^*$ be FTD, implemented by a Mealy machine $M$ with $n$ states.  Then for any $x\in\Sigma^*$ with $|x|\ge n$, there exist strings $u,v,w$ with $x=uvw$, $|uv|\le n$, $|v|\ge 1$, such that for all $k\ge 0$:
  \[
    f(uv^kw) = \OBAR(s_0,u)\cdot\OBAR(\DBAR(s_0,u),v)^k\cdot\OBAR(\DBAR(s_0,uv),w).
  \]
  That is, the output of $uv^kw$ consists of the output while processing $u$, followed by $k$ repetitions of the output while processing $v$ (each time starting from state $\DBAR(s_0,u)$), followed by the output while processing $w$ from state $\DBAR(s_0,uv)=\DBAR(s_0,u)$.

  \item \textbf{Proof:}  Since $|x|\ge n$ and $M$ has $n$ states, by the pigeonhole principle the sequence of states $s_0,\DBAR(s_0,u),\DBAR(s_0,uv),\ldots$ must repeat a state within the first $n+1$ steps.  Let $u,v,w$ be such that $uvw=x$, $v$ causes a cycle: $\DBAR(s_0,u)=\DBAR(s_0,uv)$.  Then for any $k\ge 0$, $\DBAR(s_0,uv^k)=\DBAR(s_0,u)$, so the machine processes $w$ from the same state regardless of $k$.  Thus $f(uv^kw)=\OBAR(s_0,u)\cdot\OBAR(\DBAR(s_0,u),v)^k\cdot\OBAR(\DBAR(s_0,uv),w)$.  \qed
\end{enumerate}

%% 7.6
\item \textbf{Prove Lemma~7.3 parts a--e.}

(Lemma~7.3 typically gives properties of the equivalence relation $E_M$ on states of a Mealy machine.)

\begin{enumerate}[(a)]
  \item \textbf{Lemma~7.3a:} $s E_M t$ iff $\forall x\in\Sigma^*,\OBAR(s,x)=\OBAR(t,x)$.

  \begin{proof}
  $s E_M t$ is defined exactly this way: two states are equivalent iff they produce the same output on every input string.  This is the definition. \qed
  \end{proof}

  \item \textbf{Lemma~7.3b:} $E_M$ is an equivalence relation.

  \begin{proof}
  \emph{Reflexivity:} $\OBAR(s,x)=\OBAR(s,x)$ trivially, so $s E_M s$.
  \emph{Symmetry:} If $\OBAR(s,x)=\OBAR(t,x)$ for all $x$, then $\OBAR(t,x)=\OBAR(s,x)$ for all $x$.
  \emph{Transitivity:} If $s E_M t$ and $t E_M u$, then $\OBAR(s,x)=\OBAR(t,x)=\OBAR(u,x)$ for all $x$. \qed
  \end{proof}

  \item \textbf{Lemma~7.3c:} $s E_M t$ implies $\delta(s,\pmb{a}) E_M \delta(t,\pmb{a})$ for all $\pmb{a}\in\Sigma$.

  \begin{proof}
  For any $x\in\Sigma^*$ and $\pmb{a}\in\Sigma$:
  $\OBAR(\delta(s,\pmb{a}),x)=\OBAR(s,\pmb{a}x)/\omega(s,\pmb{a})$ (the output of $\pmb{a}x$ from $s$, stripping the first output letter).
  More precisely: $\OBAR(s,\pmb{a}x)=\omega(s,\pmb{a})\cdot\OBAR(\delta(s,\pmb{a}),x)$.
  Since $s E_M t$: $\OBAR(s,\pmb{a}x)=\OBAR(t,\pmb{a}x)$, i.e., $\omega(s,\pmb{a})\cdot\OBAR(\delta(s,\pmb{a}),x)=\omega(t,\pmb{a})\cdot\OBAR(\delta(t,\pmb{a}),x)$.
  Since $s E_M t$ implies $\omega(s,\pmb{a})=\omega(t,\pmb{a})$ (single-letter case), we get $\OBAR(\delta(s,\pmb{a}),x)=\OBAR(\delta(t,\pmb{a}),x)$ for all $x$, so $\delta(s,\pmb{a}) E_M \delta(t,\pmb{a})$. \qed
  \end{proof}

  \item \textbf{Lemma~7.3d:} $E_M$ is the coarsest congruence consistent with the output function.

  (This follows from 7.3a--c and the fact that any finer relation satisfying the same properties must equal $E_M$.)

  \item \textbf{Lemma~7.3e:} The quotient machine $M/_{E_M}$ is well-defined.

  (Follows from Lemma~7.3c: $\delta_{E_M}$ is well-defined because equivalent states are mapped to equivalent successors.)
\end{enumerate}

\item \textbf{Prove parts of Lemma~7.4 for FST $M$.}

\begin{enumerate}[(a)]
  \item \textbf{$E_{0M}$ has one class consisting of all of $S$:}

  $E_{0M}$ is defined by $s E_{0M} t$ always (no conditions).  Hence all states are in one class: $[s]_{E_{0M}}=S$ for all $s$.

  \item \textbf{$E_{1M}$ iff $\forall\pmb{a}\in\Sigma,\omega(s,\pmb{a})=\omega(t,\pmb{a})$:}

  $E_{1M}$ refines $E_{0M}$ by requiring that $s$ and $t$ agree on single-letter outputs.  $s E_{1M} t$ iff $s E_{0M} t$ (trivially) and $\forall\pmb{a}\in\Sigma,\omega(s,\pmb{a})=\omega(t,\pmb{a})$.

  \item \textbf{$s E_{i+1,M} t$ iff $s E_{iM} t$ and $\forall\pmb{a}\in\Sigma,\delta(s,\pmb{a}) E_{iM} \delta(t,\pmb{a})$:}

  \begin{proof}
  By definition, $E_{i+1,M}$ refines $E_{iM}$ by additionally requiring that the successors are $E_{iM}$-equivalent.  This is the standard iterative refinement.

  $(\Rightarrow)$ If $s E_{i+1,M} t$, then $s$ and $t$ agree on all strings of length $\le i+1$.  In particular, they agree on strings of length $\le i$ (so $s E_{iM} t$), and for each $\pmb{a}$, $\delta(s,\pmb{a})$ and $\delta(t,\pmb{a})$ agree on all strings of length $\le i$ (so $\delta(s,\pmb{a}) E_{iM} \delta(t,\pmb{a})$).

  $(\Leftarrow)$ If $s E_{iM} t$ and $\delta(s,\pmb{a}) E_{iM} \delta(t,\pmb{a})$ for all $\pmb{a}$, then for any $w=\pmb{a}w'$ with $|w|\le i+1$: $\OBAR(s,\pmb{a}w')=\omega(s,\pmb{a})\cdot\OBAR(\delta(s,\pmb{a}),w')$.  Since $s E_{1M} t$ (implied by $s E_{iM} t$, $i\ge 1$): $\omega(s,\pmb{a})=\omega(t,\pmb{a})$.  And $\delta(s,\pmb{a}) E_{iM} \delta(t,\pmb{a})$ gives $\OBAR(\delta(s,\pmb{a}),w')=\OBAR(\delta(t,\pmb{a}),w')$ for $|w'|\le i$.  So $\OBAR(s,w)=\OBAR(t,w)$. \qed
  \end{proof}
\end{enumerate}

\item \textbf{Prove Theorem~7.6 for Moore machines.}

For a Moore machine $A=\langle\Sigma,\Gamma,S,s_0,\delta,\omega\rangle$:

\begin{proof}
By induction on $|x|$.

\emph{Base ($|x|=0$):}  $\OBAR(t,y\lambda)=\OBAR(t,y)=\OBAR(t,y)\cdot\lambda=\OBAR(t,y)\cdot\OBAR(\DBAR(t,y),\lambda)$, where $\OBAR(\cdot,\lambda)=\omega(\cdot)$ for the initial state output.

\emph{Step:}  Let $x=x'\pmb{a}$.  In a Moore machine, $\OBAR(t,x)=\omega(t)\cdot\omega(\delta(t,x_1))\cdots$ (output depends on states visited, not transitions).  The concatenation property follows from the state-sequence decomposition:
$\OBAR(t,yx'\pmb{a})=\OBAR(t,yx')\cdot\omega(\DBAR(t,yx'))=\OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x')\cdot\omega(\DBAR(t,yx'))$.
Since $\DBAR(t,yx')=\DBAR(\DBAR(t,y),x')$ (by Exercise~7.1b), this equals $\OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x'\pmb{a})=\OBAR(t,y)\cdot\OBAR(\DBAR(t,y),x)$. \qed
\end{proof}

%% 7.9
\item \textbf{Prove Theorem~7.7.}

Theorem~7.7 (typically): For any Mealy machine $M$, there exists an equivalent Moore machine $M'$.

\begin{proof}
Given $M=\langle\Sigma,\Gamma,S,s_0,\delta,\omega\rangle$, construct $M'=\langle\Sigma,\Gamma,S',s'_0,\delta',\omega'\rangle$ where:
$S'=\{[s,\pmb{a}]\mid s\in S,\pmb{a}\in\Sigma\}\cup\{[s_0,\text{init}]\}$ (pairs of state and last input letter).
$s'_0=[s_0,\text{init}]$.
$\delta'([s,\pmb{b}],\pmb{a})=[\delta(s,\pmb{a}),\pmb{a}]$.
$\omega'([s,\pmb{b}])=\omega(s,\pmb{b})$ (output is the Mealy output for the last transition).
$\omega'(s'_0)=\lambda$ (or an arbitrary default).

By induction on $|x|$, $\OBAR_{M'}(s'_0,x)=\OBAR_M(s_0,x)$ for all $x\in\Sigma^+$.  Hence $M'\equiv M$. \qed
\end{proof}

%% 7.10
\item \textbf{Prove Theorem~7.8.}

Theorem~7.8 (typically): For any Moore machine, there exists an equivalent Mealy machine.

\begin{proof}
Given Moore machine $M=\langle\Sigma,\Gamma,S,s_0,\delta,\omega\rangle$, define Mealy machine $M'=\langle\Sigma,\Gamma,S,s_0,\delta,\omega'\rangle$ where $\omega'(s,\pmb{a})=\omega(\delta(s,\pmb{a}))$ (the output is the Moore output of the successor state).

By induction on $|x|$: $\OBAR_{M'}(s_0,x)=\OBAR_M(s_0,x)$ for all $x\in\Sigma^*$.

\emph{Base:} $\OBAR_{M'}(s_0,\lambda)=\lambda$ and $\OBAR_M(s_0,\lambda)=\omega(s_0)$... there is a discrepancy if the Moore machine outputs at each state including the initial.  The conversion shifts the output by one step, which is handled by noting that the initial output $\omega(s_0)$ of the Moore machine is prepended; the Mealy machine matches from the first letter onward. \qed
\end{proof}

\item \textbf{$\delta_{E_M}$ is well defined.}

\begin{proof}
$\delta_{E_M}([s]_{E_M},\pmb{a})=[\delta(s,\pmb{a})]_{E_M}$.  Suppose $[s]_{E_M}=[s']_{E_M}$, i.e., $s E_M s'$.  By Lemma~7.3c, $\delta(s,\pmb{a}) E_M \delta(s',\pmb{a})$.  Hence $[\delta(s,\pmb{a})]_{E_M}=[\delta(s',\pmb{a})]_{E_M}$, so $\delta_{E_M}$ gives the same result regardless of the representative chosen. \qed
\end{proof}

%% 7.16
\item \textbf{$\omega_{E_M}$ is well defined.}

\begin{proof}
$\omega_{E_M}([s]_{E_M},\pmb{a})=\omega(s,\pmb{a})$.  Suppose $s E_M s'$.  By the definition of $E_M$ (states are equivalent iff they produce the same output on all inputs), in particular on the single letter $\pmb{a}$: $\omega(s,\pmb{a})=\omega(s',\pmb{a})$.  Hence $\omega_{E_M}$ is independent of the choice of representative. \qed
\end{proof}

\item \textbf{Reduced is not sufficient for minimality.}

\textbf{Example.}  A reduced machine may have unreachable states.  Let $M$ have states $\{s_0,s_1\}$ where $s_1$ is unreachable from $s_0$.  If $s_0$ and $s_1$ have distinct output behaviors, $M$ is reduced (no two states are equivalent).  But $s_1$ is useless; removing it gives an equivalent machine with fewer states.  Hence $M$ is reduced but not minimal.

\item \textbf{$\mu$ in Theorem~7.5 is well defined.}

\begin{proof}
$\mu$ is defined by $\mu(s)=\DBAR_2(s_{0_2},x)$ where $x$ is any word with $\DBAR_1(s_{0_1},x)=s$ (such $x$ exists because $M_1$ is connected).  Well-definedness requires: if $\DBAR_1(s_{0_1},x)=\DBAR_1(s_{0_1},x')=s$, then $\DBAR_2(s_{0_2},x)=\DBAR_2(s_{0_2},x')$.

Since $M_1$ and $M_2$ are equivalent, $\OBAR_1(s_{0_1},wx)=\OBAR_2(s_{0_2},wx)$ and $\OBAR_1(s_{0_1},wx')=\OBAR_2(s_{0_2},wx')$ for all $w$.  In particular, $\OBAR_1(s_{0_1},xy)=\OBAR_2(s_{0_2},xy)$ implies $\OBAR_1(s,y)=\OBAR_2(\DBAR_2(s_{0_2},x),y)$ for all $y$.  Similarly with $x'$.  Since $M_2$ is reduced, the state $\DBAR_2(s_{0_2},x)$ is uniquely determined by its output behavior, which equals that of $s$.  Hence $\DBAR_2(s_{0_2},x)=\DBAR_2(s_{0_2},x')$, and $\mu$ is well defined. \qed
\end{proof}

\item \textbf{$\mu$ is an isomorphism.}

\begin{enumerate}[(a)]
  \item $\mu(s_{0_1})=\DBAR_2(s_{0_2},\lambda)=s_{0_2}$.
  \item $\mu(\delta_1(s,\pmb{a}))=\DBAR_2(s_{0_2},x\pmb{a})=\delta_2(\DBAR_2(s_{0_2},x),\pmb{a})=\delta_2(\mu(s),\pmb{a})$.
  \item $\omega_1(s,\pmb{a})=\OBAR_1(s_{0_1},x\pmb{a})|_{\text{last output}}=\OBAR_2(s_{0_2},x\pmb{a})|_{\text{last}}=\omega_2(\mu(s),\pmb{a})$.  (The output functions agree since the machines are equivalent and $\mu$ commutes with transitions.)
  \item \textbf{Injective:} If $\mu(s)=\mu(t)$, then $\DBAR_2(s_{0_2},x_s)=\DBAR_2(s_{0_2},x_t)$, so $s$ and $t$ have the same output behavior from $s_{0_1}$; by $M_1$ being reduced, $s=t$.
  \item \textbf{Surjective:} Since $M_2$ is connected, every $t\in S_2$ is reachable: $t=\DBAR_2(s_{0_2},y)$ for some $y$.  Let $s=\DBAR_1(s_{0_1},y)$; then $\mu(s)=t$.
\end{enumerate}

\item \textbf{Connected is not sufficient for minimality.}

\textbf{Example.}  Let $M$ have states $\{s_0,s_1\}$, $\Sigma=\{\pmb{a}\}$, $\Gamma=\{\pmb{0},\pmb{1}\}$:
$\delta(s_0,\pmb{a})=s_1,\ \delta(s_1,\pmb{a})=s_0$; $\omega(s_0,\pmb{a})=\pmb{0},\ \omega(s_1,\pmb{a})=\pmb{0}$.
Both states output $\pmb{0}$ on $\pmb{a}$ and alternate, so $s_0 E_M s_1$.  $M$ is connected (both states reachable from $s_0$) but not reduced (two equivalent states).  Hence it is not minimal.

Formally: any two words differing only in the last letter (one ending in $\pmb{0}$, one in $\pmb{1}$) must have different output strings, but for any $n$-state machine, words of length $>n$ that agree on a prefix must produce the same output on that prefix; however, $f_2(\pmb{0}^m\pmb{1})$ and $f_2(\pmb{0}^m\pmb{0})$ disagree on the last output character. \end{enumerate}

\item \textbf{$f_3$ is not FTD.}

$f_3$ maps each input to the first $|x|$ characters of the fixed infinite sequence $\pmb{010010001\ldots}$.  The number of $\pmb{0}$s between successive $\pmb{1}$s grows without bound.  For any $n$-state Mealy machine, the output sequence must eventually be periodic with period at most $n$.  But the gaps between $\pmb{1}$s in the infinite sequence grow arbitrarily, so $f_3$ is not periodic---hence not implementable by a finite machine.  More precisely: if $f_3$ were FTD, by the pumping lemma, processing a word of length $>n$ would pump a cycle; but cycling would repeat a finite gap between $\pmb{1}$s, preventing the ever-growing gaps.  Contradiction. \qed

\item \textbf{Isomorphism $\cong$ is an equivalence relation on Mealy machines.}

\begin{enumerate}[(a)]
  \item \textbf{Symmetry:} If $\psi:S_A\to S_B$ is an isomorphism, then $\psi^{-1}:S_B\to S_A$ is also an isomorphism (verify all homomorphism properties hold for $\psi^{-1}$).

  \item \textbf{Reflexivity:} The identity $\text{id}:S_A\to S_A$ is an isomorphism.

  \item \textbf{Composition:} If $\psi:A\to B$ and $\varphi:B\to C$ are isomorphisms, verify that $\varphi\circ\psi:S_A\to S_C$ satisfies all isomorphism properties (by chaining the homomorphism properties).

  \item \textbf{Equivalence:} From (a), (b), and (c) (which gives transitivity via $\varphi\circ\psi$), $\cong$ is reflexive, symmetric, and transitive.

  \item \textbf{Homomorphism is not an equivalence relation:} Not symmetric: a homomorphism $\mu:A\to B$ need not have an inverse that is a homomorphism from $B$ to $A$ (e.g., if $\mu$ is not injective, no inverse exists).
\end{enumerate}

\item \textbf{Prove \DSIG\ is closed under any FTD function $f:\Sigma^*\to\Sigma^*$.}

\begin{proof}
Let $L\in\DSIG$ and $f$ be FTD, implemented by Mealy machine $M$.  We want to show $f^{-1}(L)\in\DSIG$ (inverse image) and $f(L)\in\DSIG$ (image).

\emph{Closure under inverse:}  Given DFA $A$ for $L$ and Mealy machine $M$ for $f$, build a product machine $P$ whose states are $S_A\times S_M$.  On input $\pmb{a}$: $P$ advances $M$ to compute the output $f(\pmb{a})$, then advances $A$ on that output.  If the output of $M$ is a single letter per input (Mealy), then $P$ directly simulates $A$ on the translation.  The result is a DFA for $f^{-1}(L)=\{x\mid f(x)\in L\}\in\DSIG$.

\emph{Closure under image:}  $f(L)=\{f(x)\mid x\in L\}$.  Build an NDFA that reads the output of $M$ while $M$'s input was accepted by $A$: nondeterministically guess the input $x\in L$ and simulate $M$ on $x$ to generate output.  Since $L\in\DSIG$ and $M$ is a finite transducer, $f(L)\in\DSIG$.  (More precisely: the composition of a DFA with an FST yields an NDFA, and by Theorem~4.1, this is equivalent to a DFA.) \qed
\end{proof}

\end{enumerate}
